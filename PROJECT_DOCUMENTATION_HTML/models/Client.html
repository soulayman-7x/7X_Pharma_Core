📝 الكود الكامل

app/models/Client.php PHP 
class Client extends Model {
    protected $table = 'clients';
    
    public function makePayment($clientId, $amount, $userId, $method = 'cash', $note = null) {
        $client = $this->getById($clientId);
        if (!$client) return false;
        $amount = min($amount, floatval($client['credit_balance']));
        if ($amount <= 0) return false;
        $sql1 = "UPDATE {$this->table} SET credit_balance = credit_balance - ? WHERE id = ?";
        $stmt1 = $this->query($sql1, [$amount, $clientId]);
        if ($stmt1->rowCount() > 0) {
            $sql2 = "INSERT INTO client_payments (client_id, amount, payment_method, note, user_id, payment_date) VALUES (?, ?, ?, ?, ?, NOW())";
            $this->query($sql2, [$clientId, $amount, $method, $note, $userId]);
            return true;
        }
        return false;
    }
    
    public function getTotalUnpaidCredit() {
        $sql = "SELECT COALESCE(SUM(credit_balance), 0) as total_unpaid FROM {$this->table}";
        $stmt = $this->query($sql);
        $result = $stmt->fetch(PDO::FETCH_ASSOC);
        return $result['total_unpaid'];
    }
    
    public function addDebt($clientId, $amount) {
        $sql = "UPDATE {$this->table} SET credit_balance = credit_balance + ? WHERE id = ?";
        $stmt = $this->query($sql, [$amount, $clientId]);
        return $stmt->rowCount() > 0;
    }
    
    public function getAll() {
        $sql = "SELECT * FROM {$this->table} ORDER BY id DESC";
        $stmt = $this->query($sql);
        if ($stmt) return $stmt->fetchAll(PDO::FETCH_ASSOC);
        return [];
    }
    
    public function getClientTransactions($clientId) {
        $sql = "SELECT * FROM client_payments WHERE client_id = ? ORDER BY payment_date DESC";
        $stmt = $this->query($sql, [$clientId]);
        if ($stmt) return $stmt->fetchAll(PDO::FETCH_ASSOC);
        return [];
    }
}

🔍 شرح الدوال بالتفصيل

🔹 makePayment — الدالة الأكثر تعقيداً

تُنفّذ عملية سداد دين عميل بخطوات منطقية محكمة قبل أي تعديل في قاعدة البيانات.

1
التحقق من وجود العميل
تستدعي getById($clientId) للتحقق من أن العميل موجود فعلاً. إذا لم يُوجد، تُعيد false فوراً.
2
تأمين المبلغ — تجنب الدفع الزائد
تستخدم min($amount, $client['credit_balance']) لضمان أن المبلغ المدفوع لا يتجاوز الرصيد الفعلي المستحق.
3
التحقق أن المبلغ موجب
إذا كان الرصيد صفراً أو سالباً (بعد min)، لا يُنفَّذ شيء وتُعاد false.
4
تحديث رصيد العميل
يُنفَّذ UPDATE clients SET credit_balance = credit_balance - ? لإنقاص الدين.
5
تسجيل الدفعة في client_payments
فقط إذا نجح التحديث (rowCount > 0)، يُسجَّل سجل الدفعة في client_payments مع التاريخ والموظف المسؤول.

⚠️ غياب Transaction — نقطة ضعف محتملة

هذه الدالة تُنفّذ عمليتَي SQL منفصلتين بدون BEGIN TRANSACTION. إذا فشل الـ INSERT الثاني بعد نجاح الـ UPDATE، سيُخفَّض رصيد العميل دون تسجيل الدفعة في السجل!

الحل المقترح: استخدام PDO Transaction لضمان atomicity:

$this->pdo->beginTransaction() → تنفيذ → $this->pdo->commit() أو $this->pdo->rollBack()

🔹 getTotalUnpaidCredit()

تُعيد مجموع جميع الديون المتراكمة لكل العملاء. تُستخدم في لوحة التحكم لعرض إحصائية الديون الإجمالية.

  • ✅ COALESCE(SUM(credit_balance), 0) يضمن إرجاع 0 بدلاً من NULL إذا كان الجدول فارغاً
  • 📊 تُستخدَم في بطاقة "إجمالي الديون" في Dashboard

🔹 addDebt($clientId, $amount)

تزيد رصيد دين العميل عند تسجيل بيع بالآجل (credit). تُستدعى من Controller الفواتير عند إتمام بيع بطريقة الدفع credit.

  • ✅ credit_balance = credit_balance + ? تُجمع على الرصيد الحالي دون الكتابة فوقه
  • ✅ تُعيد true فقط إذا تأثر صف واحد على الأقل

🔹 getAll()

تُعيد كل العملاء مرتبين من الأحدث للأقدم. تُستخدم لعرض قائمة العملاء وعند اختيار عميل في نموذج الفاتورة.

🔹 getClientTransactions($clientId)

تُعيد تاريخ دفعات عميل محدد من جدول client_payments، مرتبةً بالأحدث أولاً. تُستخدم في صفحة تفاصيل العميل.
