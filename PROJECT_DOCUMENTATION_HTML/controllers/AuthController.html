login() — معالجة بيانات الدخول

13public function login() {
14 if ($_SERVER['REQUEST_METHOD'] === 'POST') {
15 $username = trim($_POST['username']);
16 $password = $_POST['password'];
19 $userModel = $this->model('User');
21 $user = $userModel->verifyUser($username, $password);
22 if($user) {
23 $_SESSION['user_id'] = $user['id'];
24 $_SESSION['role'] = $user['role'];
25 $_SESSION['name'] = $user['name'];
26 $this->redirect('dashboard');
28 } else {
29 $this->setFlash('toast_alert', 'Incorrect username or password', 'error');
30 $this->redirect('auth');
31 }
32 }
33}
1
REQUEST_METHOD === POST — يتأكد أن الطلب قادم من الـ Form وليس زيارة مباشرة للرابط
2
trim($username) — يُزيل المسافات الزائدة من بداية ونهاية اسم المستخدم
3
verifyUser() — يبحث في DB ويتحقق من كلمة المرور بـ password_verify (آمن ضد Timing Attack)
4
نجاح — يحفظ user_id, role, name في الجلسة ويوجّه للـ dashboard
5
فشل — setFlash() يحفظ رسالة الخطأ ثم redirect لـ /auth لعرضها

⚡ لماذا redirect بعد الفشل وليس عرض الخطأ مباشرةً؟

لمنع إعادة إرسال الـ Form عند Refresh الصفحة (Post/Redirect/Get pattern). setFlash يحفظ الرسالة في SESSION ثم تُعرض في الـ redirect التالي.
